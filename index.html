Exemplos de uso do mapa

          
d3.tsv('data/pop.tsv', function(error, data) {
  if (error) {
    console.log(error);
  } else {
    var one = d3map.viz()
      .container('#example-1')
      .data(data)
      .id('UF')
      .map({
        'topojson': '/resources/br-states.min.json',
        'unit': 'states',
        'key': 'name'
      })
      .color({
        'key': '1920',
        'range': colorbrewer.YlGn[9]
      })
      .width(400)
      .height(400)
      .draw();
  }
});
          
        
          
d3.json('data/br-2.json', function(error, data) {
  if (error) {
    console.log(error);
  } else {
    var two = d3map.viz()
      .container('#example-2')
      .data(data)
      .id('id')
      .map({
        'topojson': '/resources/br-states.min.json',
        'unit': 'states',
        'key': 'name'
      })
      .color({
        'key': 'value',
        'range': ['#FEC44F', '#40004B'],
        'interpolate': true
      })
      .width(400)
      .height(400)
      .onclick(function(d) {
        console.log(d.properties.name);
      })
      .draw();
  }
});
          
        
          
d3.json('data/br.json', function(error, data) {
  if (error) {
    console.log(error);
  } else {
    var three = d3map.viz()
      .container('#example-3')
      .data(data)
      .id('id')
      .map({
        'topojson': '/resources/br-states.min.json',
        'unit': 'states',
        'key': 'name'
      })
      .color({
        'key': function(d) {
          return d.a - d.b;
        },
        'domain': [-1, 0, 1],
        'range': ['#4575B4', '#FFFFFF', '#D53E4F']
      })
      .width(400)
      .height(400)
      .draw();
  }
});
